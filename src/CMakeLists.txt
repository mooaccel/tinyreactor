find_library(
        GLOG_LIB
        NAMES glog libglog # what to look for
        HINTS "${DEPS_LIB_HINT_PATH}" # where to look

)

    message(SEND_ERROR "Did not find libglog")
endif ()

set(LIB_SRC_LINK
        ${GLOG_LIB})

target_link_libraries(tinyreactor ${LIB_SRC_LINK})
